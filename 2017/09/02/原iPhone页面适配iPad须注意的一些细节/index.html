原iPhone页面适配iPad须注意的一些细节

Sep 2, 2017

1.注意控制器上的主视图，约束不要写死高度。

2.cell自适应，参考前两篇约束文章

3.tableheaderview自适应高度
UIView *headView = [[UIView alloc] init];
headView.frame = CGRectMake(0, 0, kDeviceWidth, 0);

UIView t = [[UIView alloc] init];
t.backgroundColor = [UIColor redColor];
[headView addSubview:t];
[t mas_makeConstraints:^(MASConstraintMaker make) {
make.left.top.bottom.equalTo(headView);
make.width.equalTo(@100);
make.height.equalTo(@100);
}];

self.tableView.tableHeaderView = headView;
[headView mas_makeConstraints:^(MASConstraintMaker *make) {
make.width.equalTo(self.view.mas_width);
}];
[self.tableView.tableHeaderView layoutIfNeeded];
